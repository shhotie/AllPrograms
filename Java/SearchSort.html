Java Programs on Searching

Program for Linear Search


public class SearchLinear {
    public static int linearSearch(int[] arr, int x) {
        for (int i = 0; i < arr.length; i++) {
            if (x == arr[i]) {
                return i;
            }
        }
        return -1;
    }

    public static void main(String[] args) {
        int[] ar = {3, 46, 76, 4, 89, 7, 27};
        System.out.println(linearSearch(ar, 4));
        System.out.println(linearSearch(ar, 78));
    }
}
Output:
3
-1

Program for Binary Search


public class SearchBinary {
    public static int binarySearch(int[] arr, int x) {
        int first = 0;
        int last = arr.length - 1;
        while (first <= last) {
            int middle = (first + last) / 2;
            if (x == arr[middle]) {
                return middle;
            } else if (x > arr[middle]) {
                first = middle + 1;
            } else {
                last = middle - 1;
            }
        }
        return -1;
    }

    public static void main(String[] args) {
        int[] i = {10, 49, 67, 90, 40, 86};
        System.out.println(binarySearch(i, 49));
    }
}
Output:
1

Program for Bubble Sort


class Bubbledown {
    public static void sortdown(int[] a) {
        int n = a.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                if (a[i] > a[j]) {
                    int temp = a[i];
                    a[i] = a[j];
                    a[j] = temp;
                }
            }
        }
    }

    public static void main(String[] args) {
        int[] a = {5, 8, 1, 6, 9, 2};
        sortdown(a);
        for (int x : a) {
            System.out.println(x);
        }
    }
}
Output:
1
2
5
6
8
9
