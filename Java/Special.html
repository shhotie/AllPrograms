Program to Find if a Given Year is a Leap Year


import java.util.*;

public class Leapyear {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter the year:");
        int year = sc.nextInt();
        if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0) {
            System.out.println("It is a leap year.");
        } else {
            System.out.println("Not a leap year.");
        }
    }
}
Output:
Enter the year:
1990
Not a leap year.

Enter the year:
2016
It is a leap year.

Program to Find Days Between Two Dates


import java.util.Scanner;

class Date {
    final int[] months = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    int day, month, year;

    Date(int day, int month, int year) {
        this.day = day;
        this.month = month;
        this.year = year;
    }

    int getNumberOfLeapYears() {
        if (month > 2) {
            return year / 4 - year / 100 + year / 400;
        } else {
            return (year - 1) / 4 - (year - 1) / 100 + (year - 1) / 400;
        }
    }

    int getNumberOfDays() {
        int dayCount = year * 365 + getNumberOfLeapYears() + day;
        for (int i = 0; i < month - 1; i++) {
            dayCount += months[i];
        }
        return dayCount;
    }

    int difference(Date d1, Date d2) {
        int dy1 = d1.getNumberOfDays();
        int dy2 = d2.getNumberOfDays();
        return Math.abs(dy1 - dy2);
    }

    @Override
    public String toString() {
        return day + ":" + month + ":" + year + " ";
    }

    static Date readDate() {
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter dd:");
        int day = sc.nextInt();
        System.out.println("Enter mm:");
        int month = sc.nextInt();
        System.out.println("Enter yyyy:");
        int year = sc.nextInt();
        return new Date(day, month, year);
    }

    public static void main(String[] args) {
        Date date1 = readDate();
        Date date2 = readDate();
        System.out.println("Number of days between " + date1 + " and " + date2 + " is: " + date1.difference(date1, date2));
    }
}
